  --space-sm: 8px;
  --space-md: 12px;
  --space-lg: 16px;
  --space-xl: 24px;

  --space-gap: 6px;
  --space-inset: 10px;

  /* Typography: Size (D-06 — tokenized as-is, no value changes) */
  --text-2xs: 10px;
  --text-xs: 11px;
  --text-sm: 12px;
  --text-md: 13px;
